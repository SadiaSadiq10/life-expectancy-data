✅ 🔥 1. Education Level vs All Mortality Indicators
SELECT 
  CASE 
    WHEN schooling < 6 THEN 'Low'
    WHEN schooling BETWEEN 6 AND 10 THEN 'Medium'
    ELSE 'High'
  END AS education_level,
  AVG(life_expectancy) AS avg_life,
  AVG(adult_mortality) AS avg_adult,
  AVG(infant_deaths) AS avg_infant,
  AVG(under_five_deaths) AS avg_under5
FROM life_data
GROUP BY 
  CASE 
    WHEN schooling < 6 THEN 'Low'
    WHEN schooling BETWEEN 6 AND 10 THEN 'Medium'
    ELSE 'High'
  END;
✅ 🔥 2. Region Comparison Across All Key Factors
SELECT 
  region,
  AVG(life_expectancy) AS avg_life,
  AVG(schooling) AS avg_school,
  AVG(adult_mortality) AS avg_adult,
  AVG(gdp_per_capita) AS avg_gdp
FROM life_data
GROUP BY region
ORDER BY avg_life DESC;
✅ 🔥 3. Developed vs Developing Full Comparison
SELECT 
  economy_status_developed,
  COUNT(*) AS total_records,
  AVG(life_expectancy) AS avg_life,
  AVG(schooling) AS avg_school,
  AVG(infant_deaths) AS avg_infant
FROM life_data
GROUP BY economy_status_developed;
✅ 🔥 4. Top 10 Countries with Best Combined Score
SELECT 
  country,
  (life_expectancy + schooling - adult_mortality/10) AS score
FROM life_data
WHERE year = 2015
ORDER BY score DESC
LIMIT 10;
✅ 🔥 5. Worst Mortality Countries Despite High Education
SELECT 
  country,
  schooling,
  adult_mortality,
  life_expectancy
FROM life_data
WHERE schooling > 10
ORDER BY adult_mortality DESC
LIMIT 10;
✅ 🔥 6. GDP Range Impact on Life Expectancy
SELECT 
  FLOOR(gdp_per_capita/5000)*5000 AS gdp_range,
  AVG(life_expectancy) AS avg_life,
  AVG(schooling) AS avg_school
FROM life_data
GROUP BY FLOOR(gdp_per_capita/5000)*5000
ORDER BY gdp_range;
✅ 🔥 7. Yearly Trend of Life Expectancy and Education
SELECT 
  year,
  AVG(life_expectancy) AS avg_life,
  AVG(schooling) AS avg_school
FROM life_data
GROUP BY year
ORDER BY year;
✅ 🔥 8. Countries with Highest Improvement Over Time
SELECT 
  country,
  MAX(life_expectancy) - MIN(life_expectancy) AS life_growth,
  MAX(schooling) - MIN(schooling) AS education_growth
FROM life_data
GROUP BY country
ORDER BY life_growth DESC;
✅ 🔥 9. Mortality Burden Index (Custom Metric)
SELECT 
  country,
  (infant_deaths + under_five_deaths + adult_mortality) AS mortality_index,
  life_expectancy
FROM life_data
ORDER BY mortality_index DESC;
✅ 🔥 10. Region-wise Child Mortality vs Education
SELECT 
  region,
  AVG(infant_deaths + under_five_deaths) AS avg_child_deaths,
  AVG(schooling) AS avg_schooling
FROM life_data
GROUP BY region;
✅ 🔥 11. Countries with High GDP but Low Life Expectancy (Anomaly Detection)
SELECT 
  country,
  gdp_per_capita,
  life_expectancy,
  schooling
FROM life_data
WHERE gdp_per_capita > 15000
AND life_expectancy < 70
ORDER BY gdp_per_capita DESC;
